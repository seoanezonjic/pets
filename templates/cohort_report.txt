<%
	import os
	import pylab
	import pandas as pd
	import matplotlib.pyplot as plt

	def get_color_palette(num):
		cm = pylab.get_cmap('gist_rainbow')
		colors = []
		for i in range(num):
			color = cm(1.*i/num)  # color will now be an RGBA tuple
			colors.append(color)
		return colors

	def add_tip(text, help_text):
		return f"<div class=\"tooltip\">{text}<span class=\"tooltiptext\">{help_text}</span></div>"

	chromosome_lengths = plotter.hash_vars["chr_sizes"]
	def plot_coverage(data, plotter_list):
		data["Chr"] = pd.Categorical(data["Chr"], chromosome_lengths.keys())
		sortedd = data.sort_values(by=["Chr", "Pos"])
		chromosomes_order = sortedd["Chr"].unique()
		g = plotter_list["sns"].FacetGrid(data=sortedd, col="Chr", hue="Chr", col_wrap=2, height=1.5).map(plotter_list["plt"].plot, 'Pos', 'Count').map(plotter_list["plt"].fill_between, 'Pos', 'Count')
		for ax, chrm in zip(g.axes.flat, chromosomes_order):
				ax.axvline(x=chromosome_lengths[chrm], color='r', linestyle=':')
		return g
